# ----------- Build stage -----------
FROM maven:3.9.9-eclipse-temurin-17 AS build

WORKDIR /app

# Copy Maven descriptor first (better caching)
COPY pom.xml .
# If you use Maven wrapper, uncomment:
# COPY mvnw .
# COPY .mvn .mvn

# Download dependencies (optional optimization)
RUN mvn -q -DskipTests dependency:go-offline

# Now copy source
COPY src ./src

# Build the jar
RUN mvn -q clean package -DskipTests

# ----------- Runtime stage -----------
FROM eclipse-temurin:17-jre

WORKDIR /app

# Copy the built jar from the build stage
# This will copy whatever jar was produced in target/
COPY --from=build /app/target/*.jar app.jar

# Eureka runs on 8761 by default; Render sets PORT, but we keep 8761 as fallback
ENV PORT=8761

EXPOSE 8761

ENTRYPOINT ["java", "-jar", "app.jar"]
